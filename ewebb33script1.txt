-- create tables
CREATE TABLE product (description VARCHAR(20), cost INT, numberAvailable INT, productID INT NOT NULL, PRIMARY KEY (productID));

CREATE TABLE customer (firstName VARCHAR(30), lastName VARCHAR(30), city VARCHAR(15), phoneNumber VARCHAR(15), customerID INT NOT NULL, PRIMARY KEY (customerID), FOREIGN KEY (agentID) REFERENCES agent(agentID) ON UPDATE CASCADE);

CREATE TABLE agent (firstName VARCHAR(30), lastName VARCHAR(30), cityWorkingOut VARCHAR(15), commisionPercentage INT, agentID INT NOT NULL, PRIMARY KEY (agentID));

CREATE TABLE bought (boughtID INT NOT NULL, PRIMARY KEY(boughtID), FOREIGN KEY (customerID) REFERENCES customer(customerID), FOREIGN KEY (firstName) REFERENCES customer(firstName), FOREIGN KEY (lastName) REFERENCES customer(lastName), quantBought INT, FOREIGN KEY (productID) REFERENCES product(productID));
